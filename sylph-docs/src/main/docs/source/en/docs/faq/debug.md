## 错误捕获

```js
'use strict';

const Http = require('http');

const server = Http.createServer((req, res) => {
  const promise = new Promise((resolve, reject) => {
    let result = '';
    req.on('data', (data) => {
      result += data;
    });

    req.once('end', () => {
      const obj = JSON.parse(result);
      resolve(obj);
    });
  });

  promise.then((obj) => {
    res.writeHead(200);
    // 这里会报错
    res.end(obj.foo.bar);
  }).catch((reason) => {
    res.writeHead(500);
    res.end();
    console.error(reason);
    // 抛错后退出程序
    process.abort()
  });
});

server.listen(8080, () => {
  console.log('listening at http://localhost:8080');
});
```
