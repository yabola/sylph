```bash
sylph-bin-${version}
├── bin (启动脚本)
├── etc 
│   └── sylph (配置)
├── etl-plugins (pipeline插件,如下有两个插件包)
│   ├── flink-node-plugin
│   └── spark-node-plugin
├── jobs (job保存目录)
│   ├── a2
│   ├── spark_test1
│   └── sql_test1
├── lib (核心依赖)
├── logs
├── modules (执行器 如有flink,spark执行器)
│   ├── sylph-runner-flink
│   │   └── lib
│   └── sylph-runner-spark
│       └── lib
└── webapps (webUI)
    ├── css
    ├── fonts
    ├── img
    ├── js
    ├── ...
```
